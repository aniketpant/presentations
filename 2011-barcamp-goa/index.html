<ul class="awesome-list">
    <li><a href="#">Link #1</a></li>
    <li><a href="#">Link #2</a></li>
    <li><a href="#">Link #3</a></li>
    <li><a href="#">Link #4</a></li>
</ul>							
						

.awesome-list {
    width: 200px;
    margin: 0 auto !important; }
.awesome-list li {
    width: 100px;
    margin: 1em; padding: 0.25em;
    text-align: center;
    background: rgba(0,0,0,0.5);
    list-style: none;
    transition: all 0.25s linear; }
.awesome-list li:nth-of-type(4n+1):hover { background: royalblue; }
.awesome-list li:nth-of-type(4n+3):hover { background: #32CD32; }
.awesome-list li:nth-of-type(4n+2):hover { background: orangered; }
.awesome-list li:nth-of-type(4n):hover { background: rgba(360,2,40,1); }
.awesome-list li:hover {
    margin-left: 50px;
    color: #fff;
    transition: all 0.25s linear; }
.awesome-list li a {
    display: block;
    color: #fff;
    font-size: 1.5em;
    text-decoration: none; }
.awesome-list li a:hover, .awesome-list li a:focus { color: #fff }
